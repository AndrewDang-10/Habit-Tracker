  <!-- MAIN APP SCRIPT AND AUTH.JS IMPORT  -->
  <script type="module">
    import './auth.js';

    document.addEventListener('DOMContentLoaded', () => {
      const calendarEl = document.getElementById('calendar');
      if (!calendarEl) return;

      const calendar = new FullCalendar.Calendar(calendarEl, {
        initialView: 'dayGridMonth',
        height: 'auto',
        headerToolbar: {
          left: 'title',
          center: '',
          right: 'today prev,next'
        },
        dateClick: function (info) {
          // Let auth.js know which date was clicked
          if (window.handleDateClick) {
            window.handleDateClick(info);
          }
        },
        eventClick: function (info) {
          if (window.handleCalendarEventClick) {
            window.handleCalendarEventClick(info);
          }
        }
      });

      calendar.render();
      window.appCalendar = calendar;
      
      if (typeof window.loadHabitsToCalendar === 'function') {
        window.loadHabitsToCalendar();
      }
      if (typeof window.loadEventsToCalendar === 'function') {
        window.loadEventsToCalendar();
      }

    });
    
  </script>
